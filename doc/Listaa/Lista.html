def push_back(objeto)
    if(@Sz==0)
        @First = Nodo.new(objeto,nil,nil)
        @Last = @First
    else
        @Last.next = Nodo.new(objeto,nil,@Last)
        @Last = @Last.next
    end
    @Sz = @Sz + 1
end
