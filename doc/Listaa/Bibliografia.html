def add_author(name,apellido)
    if(@Author == nil)
        @Author = []
        @Author[0] = Listaa::Autor.new(name,apellido)
    else
        @Author << Listaa::Autor.new(name,apellido)
    end
end

def autorTo_s
    f=""
    if(@Author.count>1)
        @Author.collect { |x| f=f+x.to_s+" & "}
    else
        @Author.collect { |x| f=f+x.to_s}
    end
    f
end
